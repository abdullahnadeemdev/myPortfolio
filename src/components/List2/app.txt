import { useState } from "react";
import ListItems from "./List2/listss.jsx";
import Form from "./List2/form";

const App = () => {
  const [todos, setTodos] = useState([

    { id: 1, title: "Eat tacos", done: false },
    { id: 2, title: "Brew tea", done: false },

      </div>
    </>
  );
};

export default App;
